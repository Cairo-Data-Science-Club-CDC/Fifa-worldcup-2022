# Reading in the table from Wikipedia
url <- "https://en.wikipedia.org/wiki/FIFA_World_Cup_hosts"
# Obtain the piece of the web page that corresponds to the "wikitable" node
tbls_lst <-  url %>%
  read_html %>%
  html_table()
# Select tables of interest
tbls_lst <- tbls_lst[c(1,2,3,4,6,7,8,9,10)]

# Assign names to the tabes
tables_names <- c("List of hosts",
                  "FIFA World Cup 2006",
                  "FIFA World Cup 2010",
                  "FIFA World Cup 2014",
                  "FIFA World Cup 2018",
                  "FIFA World Cup 2022",
                  "FIFA World Cup 2026",
                  "Total bids by country",
                  "Host country performances")
names(tbls_lst) <- tolower(tables_names) %>% str_replace_all(" ","_")
# Clean columns' names
tbls_lst <- lapply(tbls_lst,  janitor::clean_names)
#
tbls_lst$total_bids_by_country <- tbls_lst$total_bids_by_country %>% 
  mutate(times_hosted = ifelse(country == "Colombia", 0, times_hosted))#colombia withdrew from hosting due to economic concens

results_order <- c("Champions",
                   "Runners-up",
                   "Third place",
                   "Fourth place",
                   "Quarter-finals",
                   "Round of 16",
                   "Second round",
                   "First round",
                   "TBD"
                   )


tbls_lst$host_country_performances  <- tbls_lst$host_country_performances %>%
  mutate(team = str_replace(team, "West Germany", "Germany"),
         result = str_replace(result, " \\(top 12\\)", ""),
         result = factor(result, levels = results_order)) %>%
  dplyr::rename(country = "team",
                years = "year")
